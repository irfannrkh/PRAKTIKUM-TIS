    // Stage 1: Filter buku dengan rating ada
    { $match: { rating: { $exists: true } } },
    
    // Stage 2: Group by publisher
    {
        $group: {
            _id: "$publisher",
            totalBuku: { $sum: 1 },
            rataRating: { $avg: "$rating" },
            totalHalaman: { $sum: "$pages" },
            rataHalaman: { $avg: "$pages" },
            daftarJudul: { $push: "$title" }
        }
    },
    
    // Stage 3: Hanya tampilkan yang totalBuku > 1
    { $match: { totalBuku: { $gt: 1 } } },
    
    // Stage 4: Sort by rataRating
    { $sort: { rataRating: -1 } },
    
    // Stage 5: Format output
    {
        $project: {
            publisher: "$_id",
            totalBuku: 1,
            rataRating: { $round: ["$rataRating", 1] },
            rataHalaman: { $round: ["$rataHalaman", 0] },
            _id: 0
        }
    }
])

.find({}, { field1: 1, field2: 1, _id: 0 })   // Include fields
.find({}, { field1: 0 })                       // Exclude fields

// 8. AGGREGATION
.aggregate([
    { $match: { filter } },                    // Filtering
    { $group: { _id: "$field", total: {$sum:1} } }, // Grouping
    { $sort: { total: -1 } },                   // Sorting
    { $project: { newField: "$oldField" } }     // Reshaping
])

// 9. UPDATE OPERATORS
{ $set: { field: value } }                     // Set value
{ $inc: { field: amount } }                     // Increment
{ $mul: { field: number } }                     // Multiply
{ $unset: { field: "" } }                       // Remove field
{ $rename: { old: "new" } }                     // Rename field

// 10. UTILITY
.countDocuments({ filter })                     // Count
.distinct("field")                               // Unique values
